# drawbar parameter

@param drawbar1
type: float
mutable: yes
step: 0.01
min: 1
max: 8
default: 8
doc: level of the sub-octave
@end

@param drawbar2
type: float
mutable: yes
step: 0.01
min: 1
max: 8
default: 8
doc: level of fifth
@end

@param drawbar3
type: float
mutable: yes
step: 0.01
min: 1
max: 8
default: 8
doc: level of reference pitch
@end

@param drawbar4
type: float
mutable: yes
step: 0.01
min: 1
max: 8
default: 1
doc: level of octave
@end

@param drawbar5
type: float
mutable: yes
step: 0.01
min: 1
max: 8
default: 3
doc: level of octave plus fifth
@end

@param percussion
type: float
mutable: no
step: 0.01
min: 0
max: 1
default: 0.8
doc: amount of percussion sound
@end

@param decay
type: float
mutable: no
step: 0.01
min: 0
max: 1
default: 0.16
doc: decay of the percussion sound
@end

@param quality
type: float
mutable: no
step: 0.01
min: 0
max: 1
default: 0.28
doc: quality of percussion sound
@end

@param volume
type: float
mutable: no
step: 0.01
min: 0

@end

type: float

# oscillators

SIN-OSC : vco1
SIN-OSC : vco2
SIN-OSC : vco3
SIN-OSC : vco4
SIN-OSC : vco5

# gain for each oscillator

VCA : gain1
VCA : gain2
VCA : gain3
VCA : gain4
VCA : gain5
VCA : mix
VCA : percgain1
VCA : percgain2
VCA : percmix

# envelope for the percussion click

ADSR : percenv
ADSR : env

# patching

vco1.out -> gain1.in
vco2.out -> gain2.in
vco3.out -> gain3.in
vco4.out -> gain4.in
vco5.out -> gain5.in
gain1.out -> mix.in
gain2.out -> mix.in
gain3.out -> mix.in
gain4.out -> mix.in
gain5.out -> mix.in

# percussion sound

vco4.out -> percgain1.in
vco5.out -> percgain2.in
percgain1.out -> percmix.in
percgain2.out -> percmix.in
percenv.out -> percmix.levelCV

# master out and envelope

percmix.out -> mix.in
env.out -> mix.levelCV

# set the gains

mix.level = param.level*0.2
gain1.level = map(param.drawbar1,0,1)
gain2.level = map(param.drawbar2,0,1)
gain3.level = map(param.drawbar3,0,1)
gain4.level = map(param.drawbar4,0,1)
gain5.level = map(param.drawbar5,0,1)

# set the detuning

vco1.detune = -1200
vco2.detune = 700
vco3.detune = 0
vco4.detune = 1200
vco5.detune = 1900

# all oscillators have the same pitch

vco1.pitch = param.pitch
vco2.pitch = param.pitch
vco3.pitch = param.pitch
vco4.pitch = param.pitch
vco5.pitch = param.pitch

# balance of the percussion sound

percgain1.level = param.quality
percgain2.level = 1-param.quality

# set the percussion envelope

percenv.attack = 0.01
percenv.decay = map(param.decay,0.05,0.4)
percenv.sustain = 0
percenv.level = param.percussion*2
percenv.release = 0.1

# set the overall envelope

env.attack = 0.01
env.decay = 0.1
env.level = param.volume*0.2
env.sustain = param.volume*0.2
env.release = 0.01

# oooh lets try and do a leslie speaker
# two rotating speakers introduce doppler effect and
# amplitude modulation
# split at 800 Hz and use delay line for the doppler
# https://shorturl.at/gknMP

LPF : lowpass
HPF : highpass
LFO : highlfo
LFO : lowlfo
DELAY : highlag
DELAY : lowlag
VCA : stereomix
VCA : highmod
VCA : lowmod
PAN : lowpan
PAN : highpan
SHAPER : distort

VCA : lowspread
VCA : highspread

mix.out -> distort.in

distort.fuzz = map(param.overdrive,0,8)

distort.out -> lowpass.in
distort.out -> highpass.in

lowpass.out -> lowlag.in
highpass.out -> highlag.in

lowpass.cutoff = 800
lowpass.resonance = 0
highpass.cutoff = 800
highpass.resonance = 0

highlfo.out -> highmod.in
highmod.out -> highlag.lagCV
lowlfo.out -> lowmod.in
lowmod.out -> lowlag.lagCV

highlfo.out -> highspread.in
highspread.out -> highpan.angleCV
lowlfo.out -> lowspread.in
lowspread.out -> lowpan.angleCV

lowlag.out -> lowpan.in
highlag.out -> highpan.in
lowpan.out -> stereomix.in
highpan.out -> stereomix.in
stereomix.out -> audio.in

# amount of stereo spread

lowpan.angle = 0
highpan.angle = 0

lowspread.level = 1
highspread.level = -1


# try around 6.8 for tremolo and 0.6 for chorale

highlfo.pitch = param.leslie_rate
lowlfo.pitch = param.leslie_rate

highlag.lag = 0.01
lowlag.lag = 0.01
highmod.level = 0.0003
lowmod.level = 0.0003

highlfo.phase = 1.57
lowlfo.phase = -1.57
